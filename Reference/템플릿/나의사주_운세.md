## 1. 나의 사주 (기본 사주팔자 해석)

```python
def get_basic_saju_interpretation(saju):
    """기본 사주 해석"""
    interpretations = []

    # 1. 일간 해석
    day_gan = saju['day_gan']
    template = db.query("""
        SELECT * FROM interpretation_template
        WHERE template_type = 'basic_saju'
        AND JSON_EXTRACT(conditions, '$.day_gan') = %s
    """, (day_gan,))
    interpretations.append(template)

    # 2. 오행 균형 해석
    오행상태 = analyze_오행균형(saju)  # "목과다", "화과다", "균형" 등
    template = db.query("""
        SELECT * FROM interpretation_template
        WHERE template_type = 'basic_saju'
        AND JSON_EXTRACT(conditions, '$.오행균형') = %s
    """, (오행상태,))
    interpretations.append(template)

    # 3. 육친 해석
    육친상태 = analyze_육친(saju)  # "비겁다", "식상다" 등
    template = db.query("""
        SELECT * FROM interpretation_template
        WHERE template_type = 'basic_saju'
        AND JSON_EXTRACT(conditions, '$.육친') = %s
    """, (육친상태,))
    interpretations.append(template)

    return interpretations

def get_yearly_fortune(year):
```
